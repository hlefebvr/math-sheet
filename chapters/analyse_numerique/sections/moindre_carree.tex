\begin{description}
\item[Définition] : Le problème des moindres carrées consiste à trouver $x^*$ telle que
    \[
        x^*=\min_{x\in\R^n}||Ax-b||
    \]
\item[Équation normale] :
    \[
        x^*=\min_{x\in\R^n}||Ax-b||
        \Leftrightarrow
        A^TAx^*=A^tb
    \]
    Cette équation provient de $||A(x^*+\Delta x)-b||_2^2$
\end{description}
